\begin{ex}%[0P1VH1-1]
[TL.100022]
Cho hàm số $y = x^2 + 2x + 1$. Vẽ đồ thị hàm số.
\begin{center}
\begin{tikzpicture}
    \draw[->] (-3,0) -- (3,0) node[right] {$x$};
    \draw[->] (0,-1) -- (0,4) node[above] {$y$};
    \draw[domain=-3:1,smooth,variable=\x,blue] plot ({\x},{\x*\x+2*\x+1});
\end{tikzpicture}
\end{center}
\choice
{$y = x^2$}
{$y = x^2 + 2x$}
{\True $y = (x+1)^2$}
{$y = x^2 - 2x + 1$}
\loigiai{
    Ta có: $y = x^2 + 2x + 1 = (x+1)^2$
    \begin{center}
    \begin{tikzpicture}
        \draw[->] (-3,0) -- (3,0);
        \draw[->] (0,-1) -- (0,4);
        \draw[red] (-1,0) circle (2pt) node[below] {$-1$};
    \end{tikzpicture}
    \end{center}
}
\end{ex}
